ALTER TABLE BOOKS ADD BESTSELLER BOOLEAN DEFAULT FALSE;


DROP FUNCTION IF EXISTS Bestseller;

DELIMITER $$

CREATE FUNCTION Bestseller(bookrents INT) RETURNS BOOLEAN DETERMINISTIC
BEGIN

   DECLARE result BOOLEAN DEFAULT FALSE;
   IF bookrents > 2 THEN
      SET result = TRUE;
   END IF;
   RETURN result;

END $$

DELIMITER ;

SELECT Bestseller(3) AS BESTSELLER;


DROP PROCEDURE IF EXISTS UpdateBestsellers;

DELIMITER $$

CREATE PROCEDURE UpdateBestsellers()
BEGIN
   DECLARE BOOKRENTS, B_ID INT;
   DECLARE FINISHED INT DEFAULT 0;
   DECLARE ALL_BOOKS CURSOR FOR SELECT B_ID FROM BOOKS;
   DECLARE CONTINUE HANDLER FOR NOT FOUND SET FINISHED = 1;
   OPEN ALL_BOOKS;
   WHILE (FINISHED = 0) DO
      FETCH ALL_BOOKS INTO B_ID;
      IF (FINISHED = 0) THEN
         SELECT COUNT(*) FROM RENTS
            WHERE BOOK_ID = B_ID
               INTO BOOKRENTS;
         UPDATE BOOKS SET BESTSELLER = Bestseller(BOOKRENTS)
            WHERE BOOK_ID = B_ID;
         COMMIT;
      END IF;
   END WHILE;
   CLOSE ALL_BOOKS;
END $$

DELIMITER ;

CALL UpdateBestsellers();
